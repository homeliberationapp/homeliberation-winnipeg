```javascript
// STEP 1: Calculate As-Is Market Value
function calculateAsIsValue(comps, property) {
  let avgPricePerSqft = comps.reduce((sum, comp) =>
    sum + (comp.price / comp.sqft), 0) / comps.length

  let baseValue = avgPricePerSqft * property.sqft

  // Adjustments
  let adjustments = 0
  if (property.hasGarage) adjustments += 8000
  if (property.hasBasement) adjustments += 15000
  if (property.condition === 'poor') adjustments -= baseValue * 0.10

  return Math.round(baseValue + adjustments)
}

// STEP 2: Your Contract Price (negotiated with seller)
let contractPrice = negotiatedPrice // User enters this

// STEP 3: Calculate Equity
let instantEquity = asIsValue - contractPrice

// STEP 4: Calculate Your Assignment Fee
let assignmentFeePercent = 0.35 // 35% default, adjustable
let yourAssignmentFee = Math.round(instantEquity * assignmentFeePercent)

// STEP 5: Buyer's Purchase Price
let buyerPurchasePrice = contractPrice + yourAssignmentFee

// STEP 6: Verify Buyer's Equity
let buyersEquity = asIsValue - buyerPurchasePrice
let buyersEquityPercent = (buyersEquity / asIsValue) * 100

// Minimum buyer equity check
if (buyersEquityPercent < 15) {
  alert("Warning: Buyer gets less than 15% equity. Reduce your fee.")
}
```
